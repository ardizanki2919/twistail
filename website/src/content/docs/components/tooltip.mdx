### TooltipContent Props

<TypeTable
  type={{
    content: {
      description: 'Content displayed inside the tooltip',

      default: undefined,
    },
    side: {
      description: 'Position of the tooltip relative to the trigger',
      type: 'top | right | bottom | left',
      default: 'top',
    },
    sideOffset: {
      description: 'Distance from the trigger',
      type: 'number',
      default: '10',
    },
    align: {
      description: 'Alignment of the tooltip relative to the trigger',
      type: 'start | center | end',
      default: 'center',
    },
    hideArrow: {
      description: 'Hide the tooltip arrow',
      type: 'boolean',
      default: 'false',
    },
    className: {
      description: 'Additional CSS class for the tooltip content',
      type: 'string',
      default: undefined,
    },
  }}
/>
